module wmc1(clk,rst,coin,lid_r,T,d_wash,soak,wash,rinse,spin,pause);
input clk,rst,coin,T,lid_r,d_wash;
output soak,wash,rinse,spin,pause;
reg [3:0]ps,ns;
parameter IDLE = 4'd0,
          SOAK = 4'd1,
          WASH = 4'd2,
          RINSE =4'd3,
          DWASH =4'd4 ,
          DRINSE =4'd5,
          SPIN = 4'd6,
          PAUSE = 4'd7;
always @ (posedge clk or negedge rst ) begin
	if (!rst)
		ps <= IDLE;
	else 
		ps <= ns;
end
always @* begin
  case (ps)
	IDLE : begin
		if ( coin )
			ns = SOAK;
	end
	SOAK : begin
		if ( T )
			ns = WASH;
		else 
			ns = SOAK;
	end
	WASH : begin
		if ( T )
			ns = RINSE;
		else
			ns = WASH;
	end
	RINSE : begin
		if ( d_wash && T )
			ns = DWASH;
		else if( !d_wash && T )
			ns = SPIN;
		else 
			ns = RINSE;
	end
	DWASH : begin
		if ( T )
			ns = DRINSE;
		else 
			ns = DWASH;
	end
	DRINSE : begin
		if ( T )
			ns = SPIN;
	end
	SPIN : begin
		if ( T )
			ns = IDLE;
		else if ( lid_r && !T )
			ns = PAUSE;
		else
			ns = SPIN;
	end
	PAUSE : begin
		if ( !lid_r )
			ns = SPIN;
	end
  default : 
	ns = IDLE;
  endcase
end
assign soak = ( ps == SOAK );
assign wash = ( ps == WASH )|( ps == DWASH );
assign rinse = ( ps == RINSE )|( ps == DRINSE );
assign spin = ( ps == SPIN );
assign pause = ( ps == PAUSE )|( ps == IDLE );

endmodule

module timer (clk,rst,T);
input clk,rst;
output [7:0]T;
reg [3:0]count;

always @ (posedge clk or negedge rst ) begin
	if (!rst)
		count <= 0;
	else begin
		count <= count + 1;
	end
end
assign T = (&count);
endmodule

module top(clk,rst,coin,lid_r,d_wash,soak,wash,rinse,spin,pause);
input clk,rst,coin,lid_r,d_wash;
output soak,wash,rinse,spin,pause;
wire T;


wmc1 UU1 (.clk(clk), .rst(rst), .coin(coin), .lid_r(lid_r), .T(T),.d_wash(d_wash), .soak(soak), .wash(wash), .rinse(rinse), .spin(spin), .pause(pause));
timer UU2 (.clk(clk), .rst(rst), .T(T));

endmodule

module tb_top;
reg clk,rst,coin,lid_r,d_wash;
wire soak,wash,rinse,spin,pause;
top UU1 (.clk(clk), .rst(rst), .coin(coin), .lid_r(lid_r), .d_wash(d_wash), .soak(soak), .wash(wash), .rinse(rinse), .spin(spin), .pause(pause));

always begin
	#5 clk = ~clk;
end

	clk = 1;
	rst = 0;
	#2 rst = 1;
	coin = 1;
	d_wash = 1;
	lid_r = 1;
	#7000 lid_r = 0;
	#1000000 $finish;
end
